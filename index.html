    <script>

        const API_URL = 'http://localhost:8000';

        let currentJobId = null;


        // Tab switching

        function showTab(tab) {

            if (tab === 'single') {

                document.getElementById('singleForm').classList.remove('hidden');

                document.getElementById('bulkForm').classList.add('hidden');

                document.getElementById('singleTab').className = 'flex-1 py-3 px-6 bg-cyan-500 rounded-lg font-semibold';

                document.getElementById('bulkTab').className = 'flex-1 py-3 px-6 bg-slate-700 hover:bg-slate-600 rounded-lg font-semibold';

            } else {

                document.getElementById('singleForm').classList.add('hidden');

                document.getElementById('bulkForm').classList.remove('hidden');

                document.getElementById('singleTab').className = 'flex-1 py-3 px-6 bg-slate-700 hover:bg-slate-600 rounded-lg font-semibold';

                document.getElementById('bulkTab').className = 'flex-1 py-3 px-6 bg-purple-500 rounded-lg font-semibold';

            }

        }


        // Single video form

        document.getElementById('singleVideoForm').addEventListener('submit', async (e) => {

            e.preventDefault();

            

            const formData = new FormData();

            formData.append('video', document.getElementById('singleVideo').files[0]);

            formData.append('website_url', document.getElementById('websiteUrl').value);

            formData.append('duration', document.getElementById('duration').value);

            formData.append('scroll_speed', document.getElementById('scrollSpeed').value);

            formData.append('video_size', document.getElementById('videoSize').value);

            formData.append('video_position', document.getElementById('videoPosition').value);


            document.getElementById('singleProgress').classList.remove('hidden');

            document.getElementById('singleResult').classList.add('hidden');


            try {

                const response = await fetch(`${API_URL}/api/generate-video`, {

                    method: 'POST',

                    body: formData

                });


                const data = await response.json();

                currentJobId = data.job_id;


                document.getElementById('singleProgress').classList.add('hidden');

                document.getElementById('singleResult').classList.remove('hidden');

                

                loadRecentVideos();

            } catch (error) {

                alert('Error: ' + error.message);

                document.getElementById('singleProgress').classList.add('hidden');

            }

        });


        // Bulk video form

        document.getElementById('bulkVideoForm').addEventListener('submit', async (e) => {

            e.preventDefault();

            

            const formData = new FormData();

            formData.append('video', document.getElementById('bulkVideo').files[0]);

            formData.append('csv_file', document.getElementById('csvFile').files[0]);

            formData.append('duration', '10');

            formData.append('scroll_speed', '100');

            formData.append('video_size', '300');

            formData.append('video_position', 'bottom-right');


            document.getElementById('bulkProgress').classList.remove('hidden');

            document.getElementById('bulkResults').classList.add('hidden');


            try {

                const response = await fetch(`${API_URL}/api/generate-bulk`, {

                    method: 'POST',

                    body: formData

                });


                const data = await response.json();

                

                document.getElementById('bulkProgress').classList.add('hidden');

                document.getElementById('bulkResults').classList.remove('hidden');

                

                const resultsList = document.getElementById('bulkResultsList');

                resultsList.innerHTML = '';

                

                data.results.forEach(result => {

                    const div = document.createElement('div');

                    div.className = 'bg-slate-900/50 rounded-lg p-4 flex justify-between items-center';

                    

                    if (result.status === 'success') {

                        div.innerHTML = `

                            <div>

                                <p class="font-semibold">${result.name}</p>

                                <p class="text-sm text-gray-400">${result.website}</p>

                            </div>

                            <a href="${API_URL}${result.download_url}" download

                               class="bg-green-500 hover:bg-green-600 px-4 py-2 rounded-lg text-sm font-semibold">

                                Download

                            </a>

                        `;

                    } else {

                        div.innerHTML = `

                            <div>

                                <p class="font-semibold">${result.name}</p>

                                <p class="text-sm text-red-400">${result.error}</p>

                            </div>

                            <span class="text-red-500">❌ Failed</span>

                        `;

                    }

                    

                    resultsList.appendChild(div);

                });


                loadRecentVideos();

            } catch (error) {

                alert('Error: ' + error.message);

                document.getElementById('bulkProgress').classList.add('hidden');

            }

        });


        // Download video

        function downloadVideo() {

            if (currentJobId) {

                window.open(`${API_URL}/api/download/${currentJobId}`, '_blank');

            }

        }


        // Load recent videos

        async function loadRecentVideos() {

            try {

                const response = await fetch(`${API_URL}/api/list-videos`);

                const data = await response.json();

                

                const container = document.getElementById('recentVideos');

                

                if (data.videos.length === 0) {

                    container.innerHTML = '<p class="text-gray-400">No videos yet</p>';

                    return;

                }

                

                container.innerHTML = '';

                data.videos.forEach(video => {

                    const div = document.createElement('div');

                    div.className = 'bg-slate-900/50 rounded-lg p-4 flex justify-between items-center';

                    div.innerHTML = `

                        <div>

                            <p class="font-semibold">${video.filename}</p>

                            <p class="text-sm text-gray-400">${video.size_mb} MB</p>

                        </div>

                        <a href="${API_URL}/api/download/${video.job_id}" download

                           class="bg-cyan-500 hover:bg-cyan-600 px-4 py-2 rounded-lg text-sm font-semibold">

                            Download

                        </a>

                    `;

                    container.appendChild(div);

                });

            } catch (error) {

                console.error('Error loading videos:', error);

            }

        }


        // Load recent videos on page load

        loadRecentVideos();

    </script>
